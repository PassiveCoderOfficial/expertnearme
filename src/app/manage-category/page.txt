/**
 * src/app/manage-category/page.tsx
 *
 * Purpose:
 * --------
 * Admin/staff-only page to manage categories:
 * - Add new categories (name, slug, optional parent)
 * - Edit and delete existing categories safely
 * - See a tree view table with parent-child hierarchy
 *
 * Notes:
 * ------
 * - This page assumes /api/categories (GET, POST) and /api/categories/[id] (PUT, DELETE) exist.
 * - Only expose this page to authorized users in production (via middleware and RBAC).
 */

'use client';

import { useEffect, useMemo, useState } from 'react';

type Cat = {
  id: number;
  name: string;
  slug: string;
  parentId: number | null;
  children: Cat[];
};

export default function ManageCategoryPage() {
  const [tree, setTree] = useState<Cat[]>([]);
  const [flat, setFlat] = useState<Cat[]>([]);
  const [loading, setLoading] = useState(false);
  const [adding, setAdding] = useState(false);

  const [newName, setNewName] = useState('');
  const [newSlug, setNewSlug] = useState('');
  const [newParentId, setNewParentId] = useState<number | null>(null);

  const [error, setError] = useState('');
  const [success, setSuccess] = useState('');

  const [editingId, setEditingId] = useState<number | null>(null);
  const [editName, setEditName] = useState('');
  const [editSlug, setEditSlug] = useState('');
  const [editParentId, setEditParentId] = useState<number | null>(null);

  const toSlug = (s: string) =>
    s
      .toLowerCase()
      .trim()
      .replace(/[^a-z0-9\s-]/g, '')
      .replace(/\s+/g, '-')
      .replace(/-+/g, '-');

  const fetchCategories = async () => {
    setLoading(true);
    try {
      const res = await fetch('/api/categories');
      const data = await res.json();
      setTree(data);

      // Flatten for parent dropdown
      const flatten = (nodes: Cat[], acc: Cat[] = [], depth = 0) => {
        nodes.forEach((n) => {
          acc.push({ ...n, name: `${'— '.repeat(depth)}${n.name}` });
          if (n.children?.length) flatten(n.children, acc, depth + 1);
        });
        return acc;
      };
      setFlat(flatten(data));
    } catch (err) {
      console.error('Failed to fetch categories:', err);
      setError('Failed to load categories');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchCategories();
  }, []);

  const handleAdd = async (e: React.FormEvent) => {
    e.preventDefault();
    setError('');
    setSuccess('');
    setAdding(true);
    try {
      const res = await fetch('/api/categories', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          name: newName.trim(),
          slug: newSlug.trim(),
          parentId: newParentId,
        }),
      });

      const data = await res.json();
      if (!res.ok) {
        setError(data?.error || 'Failed to add category');
      } else {
        setSuccess(`Category "${data.name}" added`);
        setNewName('');
        setNewSlug('');
        setNewParentId(null);
        fetchCategories();
      }
    } catch {
      setError('Network error');
    } finally {
      setAdding(false);
    }
  };

  const startEdit = (cat: Cat) => {
    setEditingId(cat.id);
    setEditName(cat.name);
    setEditSlug(cat.slug);
    setEditParentId(cat.parentId);
  };

  const handleUpdate = async () => {
    if (!editingId) return;
    setError('');
    setSuccess('');
    try {
      const res = await fetch(`/api/categories/${editingId}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          name: editName.trim(),
          slug: editSlug.trim(),
          parentId: editParentId,
        }),
      });
      const data = await res.json();
      if (!res.ok) {
        setError(data?.error || 'Failed to update category');
      } else {
        setSuccess(`Category "${data.name}" updated`);
        setEditingId(null);
        setEditName('');
        setEditSlug('');
        setEditParentId(null);
        fetchCategories();
      }
    } catch {
      setError('Network error');
    }
  };

  const handleDelete = async (id: number) => {
    if (!confirm('Delete this category?')) return;
    setError('');
    setSuccess('');
    try {
      const res = await fetch(`/api/categories/${id}`, { method: 'DELETE' });
      const data = await res.json();
      if (!res.ok) {
        setError(data?.error || 'Failed to delete category');
      } else {
        setSuccess('Category deleted');
        fetchCategories();
      }
    } catch {
      setError('Network error');
    }
  };

  const rows = useMemo(() => {
    const r: JSX.Element[] = [];
    const walk = (nodes: Cat[], depth = 0) => {
      nodes.forEach((cat) => {
        r.push(
          <tr key={cat.id}>
            <td className="border p-2">{cat.id}</td>
            <td className="border p-2">
              {editingId === cat.id ? (
                <input
                  value={editName}
                  onChange={(e) => {
                    const val = e.target.value;
                    setEditName(val);
                    setEditSlug(toSlug(val));
                  }}
                  className="border p-1 w-full"
                />
              ) : (
                <span>
                  {'\u00A0'.repeat(depth * 2)}
                  {depth > 0 && <span className="text-gray-400">↳ </span>}
                  {cat.name}
                </span>
              )}
            </td>
            <td className="border p-2">
              {editingId === cat.id ? (
                <input
                  value={editSlug}
                  onChange={(e) => setEditSlug(toSlug(e.target.value))}
                  className="border p-1 w-full"
                />
              ) : (
                cat.slug
              )}
            </td>
            <td className="border p-2">
              {editingId === cat.id ? (
                <select
                  value={editParentId ?? ''}
                  onChange={(e) =>
                    setEditParentId(e.target.value === '' ? null : Number(e.target.value))
                  }
                  className="border p-2 w-full"
                >
                  <option value="">No parent (top-level)</option>
                  {flat
                    .filter((p) => p.id !== cat.id)
                    .map((p) => (
                      <option key={p.id} value={p.id}>
                        {p.name}
                      </option>
                    ))}
                </select>
              ) : (
                cat.parentId ? flat.find((p) => p.id === cat.parentId)?.name?.replace(/^—\s*/, '') : '—'
              )}
            </td>
            <td className="border p-2 space-x-2">
              {editingId === cat.id ? (
                <>
                  <button
                    onClick={handleUpdate}
                    className="bg-green-600 text-white px-2 py-1 rounded"
                  >
                    Save
                  </button>
                  <button
                    onClick={() => {
                      setEditingId(null);
                      setEditName('');
                      setEditSlug('');
                      setEditParentId(null);
                    }}
                    className="bg-gray-400 text-white px-2 py-1 rounded"
                  >
                    Cancel
                  </button>
                </>
              ) : (
                <>
                  <button
                    onClick={() => startEdit(cat)}
                    className="bg-blue-600 text-white px-2 py-1 rounded"
                  >
                    Edit
                  </button>
                  <button
                    onClick={() => handleDelete(cat.id)}
                    className="bg-red-600 text-white px-2 py-1 rounded"
                  >
                    Delete
                  </button>
                </>
              )}
            </td>
          </tr>
        );

        if (cat.children?.length) walk(cat.children, depth + 1);
      });
    };
    walk(tree);
    return r;
  }, [tree, editingId, editName, editSlug, editParentId, flat]);

  return (
    <main className="p-8 max-w-6xl mx-auto">
      <h1 className="text-2xl font-bold mb-6">Manage Categories</h1>

      {/* Add Category Form */}
      <form onSubmit={handleAdd} className="space-y-4 mb-8">
        <div>
          <label className="block text-sm font-medium mb-1">Category Name</label>
          <input
            type="text"
            value={newName}
            onChange={(e) => {
              const val = e.target.value;
              setNewName(val);
              setNewSlug(toSlug(val));
            }}
            placeholder="e.g. Health, Legal, IT"
            className="border p-2 w-full"
            required
          />
        </div>
        <div>
          <label className="block text-sm font-medium mb-1">Slug</label>
          <input
            type="text"
            value={newSlug}
            onChange={(e) => setNewSlug(toSlug(e.target.value))}
            placeholder="health / legal / it"
            className="border p-2 w-full"
            required
          />
        </div>
        <div>
          <label className="block text-sm font-medium mb-1">Parent Category</label>
          <select
            value={newParentId ?? ''}
            onChange={(e) => setNewParentId(e.target.value === '' ? null : Number(e.target.value))}
            className="border p-2 w-full"
          >
            <option value="">No parent (top-level)</option>
            {flat.map((p) => (
              <option key={p.id} value={p.id}>
                {p.name}
              </option>
            ))}
          </select>
        </div>

        <button
          type="submit"
          disabled={adding}
          className="bg-blue-600 text-white px-4 py-2 rounded disabled:opacity-50"
        >
          {adding ? 'Adding...' : 'Add Category'}
        </button>
        {error && <p className="text-red-600 mt-2">{error}</p>}
        {success && <p className="text-green-700 mt-2">{success}</p>}
      </form>

      {/* Category Tree Table */}
      {loading ? (
        <div className="text-gray-500 italic">Loading categories...</div>
      ) : tree.length === 0 ? (
        <div className="border border-gray-300 rounded p-4 text-center text-gray-600">
          <p className="mb-2">No categories found.</p>
          <p>Use the form above to add your first parent category.</p>
        </div>
      ) : (
        <table className="w-full border-collapse border border-gray-300">
          <thead>
            <tr className="bg-gray-100">
              <th className="border p-2">ID</th>
              <th className="border p-2">Name</th>
              <th className="border p-2">Slug</th>
              <th className="border p-2">Parent</th>
              <th className="border p-2">Actions</th>
            </tr>
          </thead>
          <tbody>{rows}</tbody>
        </table>
      )}
    </main>
  );
}
